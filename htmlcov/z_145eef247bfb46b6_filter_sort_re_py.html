11def search_definite_transactions(transactions: list, searching_str: str) -> list: 

12 """Функция отфильтровывает банковские операции по графе описания""" 

13 if not transactions: 

14 raise ValueError('Список словарей пуст') 

15 else: 

16 founded_transactions = [] 

17 pattern = re.compile(searching_str) 

18 for transaction in transactions: 

19 if pattern.search(transaction.get('description', '')): 

20 founded_transactions.append(transaction) 

21 return founded_transactions

24def search_category_transactions(transactions: list, category_dict: dict) -> dict: 

25 """Функция сортирует банковские операции по графе описания и подсчитывает их количество""" 

26 if not category_dict: 

27 raise ValueError('Фильтр не задан') 

28 else: 

29 categorizied_transactions = [] 

30 for transaction in transactions: 

31 description = transaction.get('description') 

32 if description and description in category_dict.values(): 

33 categorizied_transactions.append(description) 

34 counted_categories = Counter(categorizied_transactions) 

35 return counted_categories
